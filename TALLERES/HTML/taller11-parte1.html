‚Ä¢ Implementa recursi√≥n
‚Ä¢ Caso base: n <= 0 retorna array vac√≠o
‚Ä¢ Caso recursivo: [n] + resto de n√∫meros

const contarRegresivo = function contar(n) {
    // Tu c√≥digo aqu√≠
};

// Pruebas

üìù Soluci√≥n:

        return [];
    }
    return [n].concat(contar(n - 1));
};

console.log(contarRegresivo(3)); // [3, 2, 1]

üìù Instrucciones:

‚Ä¢ Usa closure para mantener saldo privado
‚Ä¢ depositar(cantidad): suma al saldo
‚Ä¢ retirar(cantidad): resta del saldo
‚Ä¢ consultarSaldo(): retorna saldo actual

// Escribe tu funci√≥n con closure aqu√≠
function crearCuentaBancaria(saldoInicial) {
    // Tu c√≥digo aqu√≠
}

// Pruebas

üìù Soluci√≥n:

                            
function crearCuentaBancaria(saldoInicial) {
    let saldo = saldoInicial;

        depositar: function(cantidad) {
            saldo += cantidad;

        },
        consultarSaldo: function() {
            return saldo;
        }
    };
}

cuenta.depositar(500);

‚Ä¢ Retorna el promedio

// Escribe tu funci√≥n con par√°metros rest aqu√≠
function calcularPromedio(...numeros) {
    // Tu c√≥digo aqu√≠
}

// Pruebas

üìù Soluci√≥n:

                            
function calcularPromedio(...numeros) {
    let suma = 0;
    for (let numero of numeros) {
        suma += numero;
    }
    return suma / numeros.length;
}

// Pruebas

console.log(calcularPromedio(5, 15, 25, 35)); // 20
